테두리 관련 스타일(border~)

    선택자 {
      /* 테두리 스타일(종류) */
      border-style: none | solid | dotted | dashed | double | groove | ridge | inset | outset;
      /* 테두리 두께 */
      border-width: 값; /* 고정, 가변 사용 가능*/
      /* 테두리 색상 */
      border-color: 색상 정보;

      /* 위의 항목들(스타일/두께/색상) 한번에 적용 */
      border: 두께 스타일 [색상(기본 값: black)];
      
      /* 테두리 모서리 부분에 각도 적용 */
      border-radius: 값; /* 고정, 가변 사용 가능 */

      /* top, right, bottom, left 만 따로 적용 */
      border[-위치]-[속성]: 속성 값;

      /* 테두리(박스) 그림자 효과 */
      box-shadow: x y blur spread color;
    }
